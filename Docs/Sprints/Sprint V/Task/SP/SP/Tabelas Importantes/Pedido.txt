-- Criação da Tabela de Pedido
CREATE TABLE Pedido(
    pedido_id INT IDENTITY(1,1) PRIMARY KEY,
    quantidade INT,
    ativo BIT,
    cliente_id INT,
    funcionario_id INT,
    FOREIGN KEY (cliente_id) REFERENCES Cliente(cliente_Id),
    FOREIGN KEY (funcionario_id) REFERENCES Funcionarios(func_id)
);
